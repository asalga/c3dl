 16 /**
 17 	Check to see if vector 'vecArr' is valid.  That is, if it 
 18 	has the correct amount of components, is the right type and has 
 19 	the correct types.
 20 	
 21 	@param {Array} vecArr The vector to check.
 22 	
 23 	@return {boolean} True if the 'vecArr' is valid, false otherwise.
 24 */
 25 c3dl.isValidVector = function(vecArr)
 26 {
 27 	// Check if the value being passed is an array
 28 	if (vecArr instanceof Array)
 29 	{
 30 		// Need to allow 4D vectors since last element of 
 31 		// vector may not always be 1, and we can't assume
 32 		// user wants w = 1.
 33 		if (vecArr.length == 3 || vecArr.length == 4)
 34 		{
 35 			for (var i = 0; i < vecArr.length; i++)
 36 			{
 37 				// Check for bad values
 38 				if (isNaN(vecArr[i]))
 39 					return false;
 40 			}
 41 
 42 			return true;
 43 		}
 44 	}
 45 	
 46 	return false;
 47 }

 52 
 53 	@param {Array} srcVec The vector to copy.

 63 /**
 64 	Copy the components of srcVec Vector to destVec Vector.
 65 
 66 	@param {Array} srcVec The source Vector to copy from.
 67 	@param {Array} destVec The destination Vector to copy to.
 68 */
 69 c3dl.copyVectorContents = function(srcVec, destVec)
 70 {
 71 	if (!c3dl.isValidVector(srcVec) || !c3dl.isValidVector(destVec))
 72 	{
 73 		c3dl.debug.logWarning("copyVectorContents() didn't get two vectors as parameters");
 74 		return;
 75 	}
 76 	
 77 	destVec[0] = srcVec[0];
 78 	destVec[1] = srcVec[1];
 79 	destVec[2] = srcVec[2];
 80 }

 82 /**
 83 	Create a 3D Vector from the given 'newX', 'newY' 
 84 	and 'newZ' arguments.
 85 	
 86 	@param {float} newX The x value.
 87 	@param {float} newY The y value.
 88 	@param {float} newZ The z value.
 89 	
 90 	@returns {Array} A 3D Vector with the components specified by the
 91 	three arguments.
 92 */
 93 c3dl.makeVector = function(newX, newY, newZ)
 94 {
 95 	var vec = [!isNaN(newX) ? parseFloat(newX) : 0.0,
 96 						!isNaN(newY) ? parseFloat(newY) : 0.0,
 97 						!isNaN(newZ) ? parseFloat(newZ) : 0.0];
 98 	
 99 	return vec;
100 }

103 /**
104 	Normalize the given Vector. A normalized Vector points in the same direction
105 	as the original, yet has a length of 1.
106 
107 	@param {Array} vec The Vector to normalize.
108 	
109 	@returns {Array} The normalized Vector if the argument is a Vector 
110 	object, otherwise returns nulls.
111 */
112 c3dl.normalizeVector = function(vec)
113 {
114 	if (c3dl.isValidVector(vec))
115 	{
116 		var compr = vec[0] * vec[0] + vec[1] * vec[1] + vec[2] * vec[2];
117 		
118 		// Sometimes this can become invalid
119 		if (!isNaN(compr))
120 		{
121 			var ln = Math.sqrt(compr);
122 
123 			// If the length is greater then zero, return the normalized Vector
124 			if (!isNaN(ln) && ln != 0.0)
125 			{
126 				// Normalization
127 				vec[0] = vec[0] != 0.0 ? vec[0] / ln : 0.0;
128 				vec[1] = vec[1] != 0.0 ? vec[1] / ln : 0.0;
129 				vec[2] = vec[2] != 0.0 ? vec[2] / ln : 0.0;
130 				
131 				return vec;
132 			}
133 			else
134 			{
135 				c3dl.debug.logWarning('normalizeVector() called with a vector of length 0');
136 			}
137 		}
138 		else
139 		{
140 			c3dl.debug.logWarning('normalizeVector() called with a vector with compr 0');
141 		}
142 	}
143 
144 	c3dl.debug.logWarning('normalizeVector() called with a parameter that\'s not a vector');
145 	return null;
146 }

148 /**
149 	Get the dot product of two vectors.
150 	
151 	@param {Array} vecOne The first vector.
152 	@param {Array} vecTwo The second vector.
153 
154 	@returns {float} The dot product of the two specified Vectors. If 
155 	one of the Vectors was invalid, returns null.
156 */
157 c3dl.vectorDotProduct = function(vecOne, vecTwo)
158 {
159 	if (c3dl.isValidVector(vecOne) && c3dl.isValidVector(vecTwo))
160 	{
161 		return parseFloat(vecOne[0] * vecTwo[0] + vecOne[1] * vecTwo[1] + vecOne[2] * vecTwo[2]);
162 	}
163 	else
164 	{
165 		c3dl.debug.logWarning('vectorDotProduct() called with a parameter that\'s not a vector');
166 		return null;
167 	}
168 }

170 /**
171 	Get the result of projecting vector 'vecOne' onto 'vecTwo'.
172 	
173 	@param {Array} vecOne
174 	@param {Array} vecTwo
175 	
176 	@returns {Array} result of projecting vector 'vecOne' onto 'vecTwo'.
177 */
178 c3dl.vectorProject = function(vecOne, vecTwo)
179 {
180 	//           vecOne . vecTwo
181 	// result =  ---------------- x vecTwo
182 	//           vecTwo . vecTwo
183 	
184 	// get the top and bottom dot product
185 	var topDot = c3dl.vectorDotProduct(vecOne, vecTwo);
186 	var bottomDot = c3dl.vectorDotProduct(vecTwo, vecTwo);
187 
188 	return c3dl.multiplyVector(vecTwo, topDot/bottomDot);
189 }

191 /**
192 	Get the cross product of two Vectors.  The cross product is a Vector which
193 	is perpendicular to both the first and second vector.
194 	
195 	@param {Array} vecOne The first vector.
196 	@param {Array} vecTwo The second vector.
197 	
198 	@returns {Array} The cross product of 'vecOne' and 'vecTwo'.
199 */
200 c3dl.vectorCrossProduct = function(vecOne, vecTwo, dest)
201 {
202 	// Sanity Check
203 	if (c3dl.isValidVector(vecOne) && c3dl.isValidVector(vecTwo))
204 	{
205 		var thisVec = vecOne;
206 		var inVec = vecTwo;
207 		
208 		// Normalize the Units first
209 		// not sure why we need normalize them first, it would
210 		// mess up any 'pure math' calculations such as ones in picking.        
211 		//c3dl.normalizeVector(inVec);
212 		//c3dl.normalizeVector(thisVec);
213 		
214 		if (typeof(dest) == "undefined" || dest == null)
215 			dest = c3dl.makeVector();
216 		
217 		// Perform a Cross Product
218 		dest[0] = thisVec[1] * inVec[2] - thisVec[2] * inVec[1];
219 		dest[1] = thisVec[2] * inVec[0] - thisVec[0] * inVec[2];
220 		dest[2] = thisVec[0] * inVec[1] - thisVec[1] * inVec[0];
221 		
222 		return dest;
223 	}
224 	
225 	c3dl.debug.logWarning('vectorCrossProduct() called with a parameter that\'s not a vector');
226 	return null;
227 }

230 /**
231 	Get the length of the vector 'vec'.
232 	
233 	@param {Array} vec The vector for which the length is needed.
234 	
235 	@return {float} The length of the vector 'vec'.
236 */
237 c3dl.vectorLength = function(vec)
238 {	
239 	if (c3dl.isValidVector(vec))
240 	{
241 		// the dot product of a vector by itself will yeild its length.
242 		return Math.sqrt(vec[0] * vec[0] + vec[1] * vec[1] + vec[2] * vec[2]);
243 	}
244 
245 	c3dl.debug.logWarning('vectorLength() called with a parameter that\'s not a vector');
246 	return null;	
247 }

271 	Add two Vectors together and place the result in dest and 
272 	return it.
273 	
274 	@param {Array} vecOne The first Vector.
275 	@param {Array} vecTwo The second Vector.
276 	@param {Array} [dest] A vector which will hold the result.
277 	
278 	@returns {Array} The resultant Vector if both were valid 
279 	Vectors, otherwise returns null.
280 */
281 c3dl.addVectors = function(vecOne, vecTwo, dest)
282 {
283 	if (c3dl.isValidVector(vecOne) && c3dl.isValidVector(vecTwo))
284 	{
285 		if (typeof(dest) == "undefined" || dest == null)
286 			dest = c3dl.makeVector();
287 		
288 		dest[0] = vecOne[0] + vecTwo[0];
289 		dest[1] = vecOne[1] + vecTwo[1];
290 		dest[2] = vecOne[2] + vecTwo[2];
291 		
292 		return dest;
293 	}
294 	
295 	c3dl.debug.logWarning('addVectors() called with invalid parameters');
296 	return null;
297 }

329 /**
330 	Multiply the specified vector by the scalar.  Place the result
331 	in 'dest' and return it.  This operation will multiply the scalar 
332 	by each component of the vector, effectively scaling it, making it 
333 	geometrically longer or shorter.  If the scalar value is negative, 
334 	the result will point in the opposite direction of 'vec'.
335 	
336 	@param {Array} vec The vector to scale.
337 	@param {float} scalar The amount to scale the vector.
338 	@param {Array} [dest] A vector which will hold
339 	the result if provided.
340 	
341 	@returns {Array} A vector 'vec' which has been scaled by 'scalar' or
342 	returns null if 'vec' was invalid or 'scalar' was not a number.
343 */
344 c3dl.multiplyVector = function(vec, scalar, dest)
345 {
346 	if (c3dl.isValidVector(vec) && !isNaN(scalar))
347 	{
348 		if (typeof(dest) == "undefined" || dest == null)
349 			dest = c3dl.makeVector();
350 		
351 		dest[0] = vec[0] * scalar;
352 		dest[1] = vec[1] * scalar;
353 		dest[2] = vec[2] * scalar;
354 		
355 		return dest;
356 	}
357 	
358 	c3dl.debug.logWarning('multiplyVector() called with invalid parameters');
359 	return null;
360 }

363 /**
364 	Divide each component of 'vec' and store into dest and return it.
365 
366 	@param {Array} vec The vector to divide.
367 	@param {float} scalar The amount to scale 'vec'.
368 	@param {Array} [dest] A vector which will hold
369 	the result if provided.
370 
371 	@returns {Array} A vector 'vec' which has been divided by 'scalar' or
372 	returns null if 'vec' was invalid or 'scalar' was NaN.
373 */

375 {
376 	if (c3dl.isValidVector(vec) && !isNaN(scalar))

378 		if (scalar == 0)

380 			c3dl.debug.logWarning('divideVector() called with scalar 0');
381 			return null;
382 		}
383 		
384 		if (typeof(dest) == "undefined" || dest == null)
385 			dest = c3dl.makeVector();
386 		
387 		dest[0] = vec[0] / scalar;
388 		dest[1] = vec[1] / scalar;
389 		dest[2] = vec[2] / scalar;
390 		
391 		return dest;
392 	}
393 	
394 	c3dl.debug.logWarning('divideVector() called with invalid parameters');
395 	return null;

432 /**
433 	Compare two vectors for equality. Two Vectors are said to be equal 
434 	if all their components are equal.
435 	
436 	@param {Array} vecOne Vector one.
437 	@param {Array} vectwo Vector two.
438 	
439 	@return {boolean} True if both vectors are equal, null otherwise.
440 */
441 c3dl.isVectorEqual = function(vecOne, vecTwo)
442 {
443 	// Sanity Check
444 	if (c3dl.isValidVector(vecOne) && c3dl.isValidVector(vecTwo))
445 	{
446 		// add tolerance to calculations
447 		return (vecOne[0] == vecTwo[0] && vecOne[1] == vecTwo[1] && vecOne[2] == vecTwo[2]);
448 	}
449 	
450 	c3dl.debug.logWarning('isVectorEqual() called with invalid parameters');
451 	return null;

456 /**
457 	Check to see if a Vector is a zero vector, that is a vector with a 
458 	length of zero, or it has components close enough to zero to be 
459 	considered zero.  'Close enough' is considered true if the components
460 	lie between the -TOLERANCE and +TOLERANCE constant.
461 	
462 	@param {Array} vec The Vector to check.
463 	
464 	@return {boolean} True if the vector is considered a zero vector, false otherwise.
465 */
466 c3dl.isVectorZero = function(vec)
467 {
468 	if (c3dl.isValidVector(vec))
469 	{
470 		// Check for a tolerance
471 		return ((-c3dl.TOLERANCE < vec[0] && vec[0] < c3dl.TOLERANCE) && 
472 				(-c3dl.TOLERANCE < vec[1] && vec[1] < c3dl.TOLERANCE) && 
473 				(-c3dl.TOLERANCE < vec[2] && vec[2] < c3dl.TOLERANCE));
474 	}
475 	
476 	return false;
477 }

480 /**
481 	Get the angle (in degrees) between two vectors.
482 
483 	@param {Array} vecOne The first vector.
484 	@param {Array} vecTwo The second vector.
485 
486 	@return {float} The angle in degrees between the two vectors.
487 */
488 c3dl.getAngleBetweenVectors = function(vecOne, vecTwo)
489 {
490 	var dot = c3dl.vectorDotProduct(vecOne, vecTwo);
491 	return c3dl.radiansToDegrees(Math.acos(dot));
492 }
